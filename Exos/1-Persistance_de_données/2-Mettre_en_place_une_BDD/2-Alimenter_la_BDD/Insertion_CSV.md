# Importer CSV
```bash
    LOAD DATA LOCAL INFILE 'produit.csv'
    INTO TABLE produit
    FIELDS TERMINATED BY ';' 
    LINES TERMINATED BY '\r\n'
    IGNORE 1 LINES
    (CODART,LIBART,UNIMES,STKALE,STKPHY,QTEANN);


    LOAD DATA LOCAL INFILE 'vente.csv'
    INTO TABLE produit
    FIELDS TERMINATED BY ';' 
    LINES TERMINATED BY '\r\n'
    IGNORE 1 LINES
    (CODART,NUMFOU,DELLIV,QTE1,PRIX1,QTE2,PRIX2,QTE3,PRIX3);
```
